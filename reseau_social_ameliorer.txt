contenu de script.js  a la base du site 

document.addEventListener('DOMContentLoaded', () => {
    const links = document.querySelectorAll('.social-menu ul li a');

    links.forEach(link => {
        link.addEventListener('mouseover', () => {
            link.style.backgroundColor = '#777';
        });

        link.addEventListener('mouseout', () => {
            link.style.backgroundColor = '';
        });
    });
});
(function () {
    var url = window.location.protocol + '//' + window.location.host + window.location.pathname;

    // Supprimer index.html
    url = url.replace(/index\.html$/i, '');

    // Créer la balise canonique
    var link = document.createElement('link');
    link.rel = 'canonical';
    link.href = url;

    document.head.appendChild(link);
})();
(function () {
    var path = window.location.pathname;
    var page = path.substring(path.lastIndexOf('/') + 1) || 'Accueil';

    page = page.replace('.html', '')
               .replace(/[-_]/g, ' ')
               .replace(/\b\w/g, function (l) { return l.toUpperCase(); });

    document.title = page + ' – Le site de Pascal Delfosse - Artiste Contemporain';
})();
